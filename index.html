searchInput.value.toLowerCase() : ''; filterCards(searchTerm, category); }); }); } // Main Filter Function function filterCards(searchTerm, category) { let visibleCount = 0; serviceCards.forEach(card => { const title = card.querySelector('h3').textContent.toLowerCase(); const description = card.querySelector('p').textContent.toLowerCase(); const cardCategory = card.getAttribute('data-category'); // Check if title or description matches search term const matchesSearch = title.includes(searchTerm) || description.includes(searchTerm); // Check if category matches (or if 'all' is selected) const matchesCategory = category === 'all' || cardCategory === category; if (matchesSearch && matchesCategory) { card.classList.remove('hidden'); visibleCount++; } else { card.classList.add('hidden'); } }); // Handle No Results if (visibleCount === 0) { if (!servicesGrid.querySelector('.no-results')) { const noResults = document.createElement('div'); noResults.className = 'no-results'; noResults.innerHTML = `

No Services Found

Try adjusting your search or filter criteria.

`; servicesGrid.appendChild(noResults); } } else { const noResults = servicesGrid.querySelector('.no-results'); if (noResults) { noResults.remove(); } } } } ```
